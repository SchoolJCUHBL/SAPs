int l = N + 1

for (int i = 0, i < (0,5 * N + 2), i++)
{
	grid[i * l] = true;
	grid[l - 1 + i * l] = true;
}

for (int i = 0, i < l, i++)
{
	grid[i] = true;
	grid[0,5 * N * l - i] = true;
}

for (int i = 0, i < N - 1, i++)
{
	grid[2 * l - 1 - i] = true;
}


N = SAPlengte
k = geheel positief getal
l = hoogte = N + 1
lengte = 0,5 * N + 2
beginpunt = (1 , 0,5 * N + 1) -> l + 0,5 * N + 1

0 + k*l (ondergrens)
l - 1 + k*l (bovengrens)
0 t/m l (linkergrens)
0,5 * N * l - 1 - l t/m 0,5 * N * l - 1(rechtergrens)
l + 0,5 * N + 2 t/m 2 * l - 1 (boven beginpunt)
